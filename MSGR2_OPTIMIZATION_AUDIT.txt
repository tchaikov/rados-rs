  // Helper: build secure frame payload (no CRCs, alignment padding + epilogue)
  fn build_secure_payload(&self, frame: &Frame) -> Result<Bytes> {
      const BLOCK_SIZE: usize = 16;
      // Calculate exact size: sum(segment_len + padding) + epilogue_for_multi
      let mut total = 0;
      for seg in &frame.segments {
          if !seg.is_empty() {
              total += (seg.len() + BLOCK_SIZE - 1) & !(BLOCK_SIZE - 1);
          }
      }
      if frame.preamble.num_segments > 1 {
          total += BLOCK_SIZE;
      }
      
      // Build in one pass, pre-allocated
      let mut payload = BytesMut::with_capacity(total);
      for seg in &frame.segments {
          if !seg.is_empty() {
              payload.extend_from_slice(seg);
              let aligned = (seg.len() + BLOCK_SIZE - 1) & !(BLOCK_SIZE - 1);
              payload.put_bytes(0, aligned - seg.len());
          }
      }
      if frame.preamble.num_segments > 1 {
          payload.put_u8(0x0e);  // FRAME_LATE_STATUS_COMPLETE
          payload.put_bytes(0, BLOCK_SIZE - 1);
      }
      Ok(payload.freeze())
  }

INVARIANTS TO PRESERVE
================================================================================

Preamble (32 bytes):
  • Encoded: tag, num_segments, 4× segment descriptors, flags, CRC
  • Plaintext: CRC covers preamble data itself
  • Encrypted: included in GCM-encrypted block

Segments:
  • In order: seg0, seg1, seg2, seg3
  • Each has logical_len (actual payload size) in descriptor
  • Empty trailing segments allowed

Frame Assembly Variants:
  • Plaintext rev0: preamble + segments + epilogue(late_flags + 4 CRCs)
  • Plaintext rev1: preamble + seg0 + CRC + seg1..3 + epilogue(late_status + 3 CRCs)
  • Secure mode:   preamble + [seg0_padded | seg1_padded | ... | epilogue_padded]
                   (no CRCs, 16-byte alignment, GCM authentication tag)

Secure Mode Inline Optimization:
  • First 48 bytes of payload encrypted with preamble (80 → 96 bytes with GCM tag)
  • Remaining encrypted separately if any (N → N+16 bytes with GCM tag)
  • Epilogue: 16 bytes for multi-segment, 0 bytes for single-segment
